饼干

2021-12-21

这道题我们首先需要发现一个性质：贪婪度越大的孩子获得的饼干数越大

这个使用微扰法贪心可以非常容易地证明，这里不再赘述

设 fi,jf_{i , j}fi,j​ 表示所有将 jjj 个饼干分配给前 iii 个小朋友的方案中的最小怒气值

根据经验我们会考虑最后一个小朋友分配多少饼干，但是我们无法高效地进行状态转移。这个我用区间划分试过，失败了。区间划分的话我需要考虑最后一个小朋友获得了多少饼干才能转移

那，能不能压掉最后一个小朋友获得的饼干数的那一维呢？其实是可以的

我们发现，对于当前考虑的第 iii 个孩子，如果 ai≥2a_i ≥ 2ai​≥2，那么完全可以让前 iii 个孩子全都少得到一个饼干，不影响最终结果，这里用到了一个相对不变性

如果 ai=1a_i = 1ai​=1 ，我们不放考虑最后有几个小朋友获得了 111 个饼干

因此，有

fi,j=min(fi,j−i,fk,j−(i−k)+k∗(sumi−sumk))，k∈[1,i−1]f_{i , j} = min(f_{i , j - i} , f_{k , j - (i - k)} + k * (sum_i - sum_k))，k ∈[1 , i - 1] fi,j​=min(fi,j−i​,fk,j−(i−k)​+k∗(sumi​−sumk​))，k∈[1,i−1]

回过来看这道题，启示我们，做题的时候首先要观察有没有什么特殊的性质，可能会用到贪心等等。这道题就是发现了 aia_iai​ 随 gig_igi​ 单调递减的性质。

有时候会用等效的手法对状态空间进行放缩。这道题，我们发现最终影响由相对的大小决定而于 aia_iai​ 没有太大的关系，因此我们考虑让每个小朋友少拿一个饼干，结果不会改变。正是发现了相对不变性质，才压缩掉了枚举最小值的那一维，直接枚举有多少小朋友获得了一个饼干

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 35;
const int M = 5e3 + 10;

int n , m , g[N] , b[N] , ans[N];
long long f[N][M] , sum[N];

struct node{
    int i , j;
    node(){}
    node(int _i , int _j){
        i = _i , j = _j;
    }
}pre[N][M];

struct data{
    int id , v;
    bool operator < (const data &x)const{
        return x.v < v;
    }
}a[N];

int main(void){

    scanf("%d%d" , &n , &m);

    for(int i = 1 ; i <= n ; ++ i){
        scanf("%d" , &a[i].v);
        //sum[i] = sum[i - 1] + a[i];
        a[i].id = i;
    }

    sort(a + 1 , a + 1 + n);

    for(int i = 1 ; i <= n ; ++ i){
        sum[i] = sum[i - 1] + a[i].v;
    }

    for(int i = 0 ; i <= n ; ++ i){
        for(int j = 0 ; j <= m ; ++ j){
            f[i][j] = (1ll << 50);
        }
    }

    f[0][0] = 0;

    for(int i = 1 ; i <= n ; ++ i){
        for(int j = i ; j <= m ; ++ j){
            for(int k = 0 ; k < i ; ++ k){
                if(f[k][j - (i - k)] + k * (sum[i] - sum[k]) < f[i][j]){
                    f[i][j] = f[k][j - (i - k)] + k * (sum[i] - sum[k]);
                    pre[i][j] = node(k , j - (i - k));
                }
            }
            if(f[i][j - i] < f[i][j]){
                f[i][j] = f[i][j - i];
                pre[i][j] = node(i , j - i);
            }
        }
    }

    int x = n , y = m;

    while(x && y){
        int tx = x , ty = y;
        if(pre[x][y].i == x && pre[x][y].j == y - x){
            b[1] ++ , b[x + 1] -- ;
            x = pre[tx][ty].i , y = pre[tx][ty].j;
        }
        else {
            int k = pre[x][y].i;
            b[k + 1] ++ , b[x + 1] -- ;
            x = pre[tx][ty].i , y = pre[tx][ty].j;
        }
    }

    for(int i = 1 ; i <= n ; ++ i){
        b[i] += b[i - 1];
    }
    
    for(int i = 1 ; i <= n ; ++ i){
        ans[a[i].id] = b[i];
    }

    printf("%lld\n" , f[n][m]);
    
    for(int i = 1 ; i <= n ; ++ i){
        printf("%d " , ans[i]);
    }

    return 0;

}
