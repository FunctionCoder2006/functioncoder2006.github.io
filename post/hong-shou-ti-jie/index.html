单调队列 差分

触手题解

Posted on 2021-11-14，3 min read

先考虑第一问

我们发现，长度 xxx 是不变的，而某一个柱子又取决于所有能覆盖 tatata 的区间的最小值，因此考虑单调队列。

但是有一些柱子被刷过了，但是在新的区间里有可能会被更新成更大的值。

我们也不难发现，新的区间和上一个区间的交集长度是不变的，而且因此考虑使用差分数组维护 deltadeltadelta

然后考虑第二问

不难发现，最终柱子的颜色高度一定会形成若干个高度相等连续的区间，因此对于每一个连续的区间，能一下刷 xxx 就刷 xxx 个，如果还剩下的话就需要分类讨论了。如果这个区间的值是左中右最大的，那么可以将原来刷过的柱子再刷一遍，顺便把剩下的也刷了，需要多刷一次；如果这个区间不是左中右最大的，那么剩下的柱子就可以和比它大的柱子刷掉。

因此，对于一个连续区间，如果长度为 nnn 的话，至少需要刷 ceil(n/x)ceil(n / x)ceil(n/x) 次

后来还证明了一个连续区间一定会被长度为 xxx 的区间覆盖掉

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>
#define int long long

using namespace std;

const int N = 2e5 + 10;

int n , m , a[N] , q[N] , hh = 1 , tt = 0 , h[N] , Last , b[N];
int ans1 , ans2;

inline int calc(int l , int r){
    int Len = r - l + 1;
    if(Len % m == 0)return Len / m;
    return Len / m + 1;
}

signed main(void){

    freopen("xyx.in" , "r" , stdin);
    freopen("xyx.out" , "w" , stdout);

    scanf("%lld%lld" , &n , &m);

    for(int i = 1 ; i <= n ; ++ i){
        scanf("%lld" , &a[i]);
    }

    for(int i = 1 ; i <= m ; ++ i){
        while(hh <= tt && a[q[tt]] >= a[i])
            -- tt; 
        q[++ tt] = i; 
    }

    b[1] += a[q[hh]] , b[m + 1] -= a[q[hh]];

    Last = a[q[hh]]; 

    for(int i = m + 1 ; i <= n ; ++ i){
        while(hh <= tt && a[q[tt]] >= a[i])
            -- tt;
        while(hh <= tt && q[hh] + m <= i)
            ++ hh;
        q[++ tt] = i;
        int delta = max(a[q[hh]] - Last , 0ll);
        b[i - m + 1] += delta , b[i] -= delta;
        b[i] += a[q[hh]] , b[i + 1] -= a[q[hh]];
        Last = a[q[hh]];
    }  

    for(int i = 1 ; i <= n ; ++ i){
        b[i] += b[i - 1];
        h[i] = b[i];
        ans1 += h[i];
    }

    int last = -1;

    int l = 1 , r = 1;
    while(1){
        while(r + 1 <= n && h[r + 1] == h[r])
            ++ r;
        ans2 += calc(l , r);
        ++ r;
        if(r > n)break;
        l = r;
    }

    printf("%lld\n%lld" , ans1 , ans2);

    return 0;   

}
