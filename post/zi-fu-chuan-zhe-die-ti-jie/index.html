字符串折叠题解

2021-10-22 5 min read

这道题是自己独立做出来的，甚是开心。

首先我们考虑字符串折叠有哪些方式，一种是并列式比如 2(ABC)5(KFD)2(ABC)5(KFD)2(ABC)5(KFD) 另一种是嵌套式，比如 3(9(HAH))3(9(HAH))3(9(HAH)) 需要注意的是，嵌套式括号里边也可以使用并列式，比如 2(3(AB)4(HB))2(3(AB)4(HB))2(3(AB)4(HB)) 并列式括号里边自然也可以用嵌套式（之前想当然了，以为把区间 [L,R][L , R][L,R] 折叠就完事儿了，没想到其还可以继续折叠……）

当看到并列式，你是否想起了区间 dpdpdp fl,r=min(fl,k+fk+1,j)f_{l , r} = min(f_{l , k} + f_{k + 1 , j})fl,r​=min(fl,k​+fk+1,j​) 这个式子？就是把区间 [L,R][L , R][L,R] 拆成两个区间，对两个区间分别折叠然后合并。

那嵌套式怎么办？其实不难想到，嵌套式一定是多个循环节拼凑成的，也就是说，对于一个区间 [L,R][L , R][L,R] 我们考虑枚举 tatata 的循环节，然后对于每一个循环节，继续进行字符串折叠。

有一个坑点： 区间[L,R][L , R][L,R] 也是 tatata 的循环节，而如果我们对区间 [L,R][L , R][L,R] 进行递归的话，我们相当于什么都没有做，此时如果区间 [L,R][L , R][L,R] 的最短循环节是 tatata 本身的话，就会进入死循环。

你可能会问，如果出现了 ABBABBABBABBABBABBABBABBABBABBABBABBABBABBABB时，你将其处理成并列式的时候，将其折叠成 2(AB)3(AB)2(AB)3(AB)2(AB)3(AB) 这不是浪费吗？

是这样的虽然说将其变成并列式有些浪费，但是我同时也对 tatata 进行了嵌套式折叠，这样一来，取一个 minminmin 就可以找到最优解。


#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
#include<cstring>
#include<vector>

using namespace std;

const int N = 120;
const int base = 131;
const int mod1 = 1e9 + 9 , mod2 = 998244353;

char str[N];
int n;
long long s1[N] , s2[N] , p1[N] , p2[N] , f[N][N];

vector<vector<int> > vec(N);

long long solve(long long s[] , long long p[] , int mod , int l , int r){
    return (s[r] - (s[l - 1] * p[r - l + 1] % mod) + mod) % mod;
}

inline int num(int x){
    if(x <= 9)return 1;
    if(10 <= x && x <= 99)return 2;
    return 3;
}

inline int dfs(int l , int r){
    
    if(f[l][r])return f[l][r];

    int Len = r - l + 1;
    int res = Len;

    if(l == r)return 1;
    if(l > r)return 0;

    for(int i = 0 ; i < vec[Len].size() ; ++ i){
        if(vec[Len][i] == Len)continue;
        if(solve(s1 , p1 , mod1 , l + vec[Len][i] , r) == solve(s1 , p1 , mod1 , l , r - vec[Len][i])
            && solve(s2 , p2 , mod2 , l + vec[Len][i] , r) == solve(s2 , p2 , mod2 , l , r - vec[Len][i])){
                res = min(res , 2 + num(Len / vec[Len][i]) + dfs(l , l + vec[Len][i] - 1));
            }
    }

    for(int k = l ; k < r ; ++ k){
        res = min(res , dfs(l , k) + dfs(k + 1 , r));
    }

    return f[l][r] = res;
}

int main(void){

    scanf("%s" , str + 1);

    n = strlen(str + 1);

    for(int i = 1 ; i <= n ; ++ i){
        for(int j = i ; j <= n ; j += i){
            vec[j].push_back(i);
        }
    }

    p1[0] = p2[0] = 1;
    
    for(int i = 1 ; i < N ; ++ i){
        p1[i] = p1[i - 1] * base % mod1;
        p2[i] = p2[i - 1] * base % mod2;
    }

    for(int i = 1 ; i <= n ; ++ i){
        s1[i] = (s1[i - 1] * base + str[i]) % mod1;
        s2[i] = (s2[i - 1] * base + str[i]) % mod2;
    }

    cout << dfs(1 , n) << endl;

    return 0;
}
