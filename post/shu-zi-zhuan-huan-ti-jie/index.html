数字转换题解

2021-11-17 2 min read # 树的直径 # 树形dp # 动态规划

这道题乍一看是换根 dpdpdp，其实用换根 dpdpdp 做也是可以的，因为这道题就是从一个点出发往下走最远能走到多远，没有意识到这就是在求树的直径，树的直径就是树的最长路径也就是任意两点距离的最大值

说到底还是对树的直径概念模糊

这道题还用到了一个求约束和的方法，这个自己会但是第一反应不是 tatata ，因为自己没有注意到是将 111 ~ nnn 的约束和都求出来

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 1e5 + 10;

int n , sum[N] , d[N] , ans;

int head[N << 1] , nx[N << 1] , ver[N << 1] , idx;
inline void add(int x , int y){
    ver[idx] = y;
    nx[idx] = head[x] , head[x] = idx ++ ;
    return;
}  

inline void dfs(int u , int fa){
    for(int i = head[u] ; ~i ; i = nx[i]){
        int v = ver[i];
        if(v == fa)continue;
        dfs(v , u);
        ans = max(ans , d[u] + d[v] + 1);
        d[u] = max(d[u] , d[v] + 1);
    }
    return;
}

int main(void){

    scanf("%d" , &n);

    for(int i = 1 ; i <= n ; ++ i){
        for(int j = i ; j <= n ; j += i){
            sum[j] += i;
        }
    }

    memset(head , -1 , sizeof(head));

    for(int i = 1 ; i <= n ; ++ i){
        sum[i] -= i;
        if(sum[i] < i){
            add(i , sum[i]);
            add(sum[i] , i);
        }
    }

    dfs(1 , -1);

    printf("%d" , ans);

    return 0;

}
