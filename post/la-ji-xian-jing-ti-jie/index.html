垃圾陷阱题解

2021-11-14 3 min read # 背包 # 动态规划

这道题有两种做法，一种是将高度作为状态，一种是将生命作为状态，这里我选择了用生命做状态

转移方程也很好写，但是就是有一个点需要注意

由于有可能牛还没等到垃圾就饿死了，但是 dpdpdp 过程中，可能会出现牛已经饿死了，但是吃了垃圾又活了的情况。

所以，考虑改变状态的含义，设 fi,jf_{i , j}fi,j​ 表示所有只考虑前 iii 个垃圾，牛已经经过了等待第 iii 个垃圾降临的时间，也即经过了 a[i].t−a[i−1].ta[i].t - a[i - 1].ta[i].t−a[i−1].t 时间后生命为 jjj 的方案中，能撘的高度的最大值

如果是前 i−1i - 1i−1 个垃圾，那么 jjj 就得变成 j+(a[i].t−a[i−1].t)j + (a[i].t - a[i - 1].t)j+(a[i].t−a[i−1].t)，因为这样，jjj 的含义就得变成“经过了第 i−1i - 1i−1 垃圾的降临时间”

#include<bits/stdc++.h>

using namespace std;

const int N = 110;
const int M = 3010;

int maxh , n , f[N][M];

struct node{
    int t , l , h;
}a[N];

bool cmp(node x , node y){
    return x.t < y.t;
}

int main(void){

    scanf("%d%d" , &maxh , &n);

    for(int i = 1 ; i <= n ; ++ i){
        scanf("%d%d%d" , &a[i].t , &a[i].l , &a[i].h);
    }

    sort(a + 1 , a + 1 + n , cmp);

    memset(f , 0xcf , sizeof(f));

    f[0][10] = 0;

    for(int i = 1 ; i <= n ; ++ i){
        for(int j = 3000 ; j >= 0 ; -- j){
            f[i][j] = max(f[i][j] , f[i - 1][j + (a[i].t - a[i - 1].t)] + a[i].h);
            f[i][j + a[i].l] = max(f[i][j + a[i].l] , f[i - 1][j + (a[i].t - a[i - 1].t)]);
            if(f[i][j] >= maxh){
                printf("%d" , a[i].t);
                return 0;
            }
        }
    }

    int Life = 10 , ans = 0;

    for(int i = 1 ; i <= n ; ++ i){
        if(Life < (a[i].t - a[i - 1].t)){
            ans += Life;
            Life = 0;
            break;
        }
        Life -= (a[i].t - a[i - 1].t);
        Life += a[i].l;
        ans += (a[i].t - a[i - 1].t);
    }

    if(Life)ans += Life;

    printf("%d" , ans);

    return 0;

}
