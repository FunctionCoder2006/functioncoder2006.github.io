#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int M = 4e6 + 10;
const int N = 1e5 + 10;

int n , m , t[N] , f[M];
int cnt[M] , sum[M] , maxt , ans = 2e9;

int main(void){

    scanf("%d%d" , &n , &m);

    for(int i = 1 ; i <= n ; ++ i){
        scanf("%d" , &t[i]);
        maxt = max(maxt , t[i]);
        cnt[t[i]] ++ ;
        sum[t[i]] += t[i];
    }

    for(int i = 1 ; i <= maxt + m ; ++ i){
        cnt[i] += cnt[i - 1];
        sum[i] += sum[i - 1];
    }

    for(int i = 0 ; i < m ; ++ i){
        f[i] = i * cnt[i] - sum[i];
    }

    for(int i = m ; i <= maxt + m ; ++ i){
        f[i] = i * cnt[i] - sum[i];
        if(cnt[i] == cnt[i - m]){
            f[i] = f[i - m];
            continue;
        }
        for(int j = max(0 , i - 2 * m + 1) ; j <= i - m ; ++ j){
            f[i] = min(f[i] , f[j] + i * (cnt[i] - cnt[j]) - (sum[i] - sum[j]));
        }   
    }

    for(int i = maxt ; i <= maxt + m ; ++ i){
        ans = min(ans , f[i]);
    }

    printf("%d" , ans);

    return 0;

}
