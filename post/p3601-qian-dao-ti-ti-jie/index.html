P3601 签到题题解

Posted on 2021-10-10，3 min read

这道题首先转化qiaodaoqiaodaoqiaodao这个函数，很明显，111~nnn里的整数除去与nnn互质的数剩下的数就是与其不互质的数，故qiandao(i)qiandao(i)qiandao(i) = i−φ(i)i - φ(i)i−φ(i)

然后我们考虑区间筛然后求φφφ

几个坑点：

1.1.1.我们区间筛的时候筛的是111~R\sqrt{R}R​的质因数，但当我们求φ(n)φ(n)φ(n)的时候我们需要枚举到ta的所有质因数，而ta的质因数有可能超过n\sqrt{n}n​，所以我们还需要记录restrestrest数组来模拟分解质因数，如果我们跑完一遍埃氏筛之后某一个数nnn的restrestrest值大于1，就说明ta有大于n\sqrt{n}n​的质因子，我们就需要更新ta的φφφ值然后再累计贡献。

2.2.2.在给φ和restφ和restφ和rest初始化的时候一定要从lll枚举到rrr，千万不要从111枚举到r−l+1r - l + 1r−l+1因为我们赋值的时候是令其φφφ值等于iii。

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>
#define int long long

using namespace std;

const int N = 1e6 + 10;
const int mod = 666623333;

int l , r , cnt , mindiv[N] , prime[N];
int phi[N] , rest[N];

inline void Prime(int n){
    for(int i = 2 ; i <= n ; ++i){
        if(!mindiv[i]){
            mindiv[i] = i;
            prime[++ cnt] = i;
        }
        for(int j = 1 ; j <= cnt ; ++j){
            if(prime[j] > n / i || prime[j] > mindiv[i])
                break;
            mindiv[i * prime[j]] = prime[j];
            if(i % prime[j] == 0)
                break;
        }
    }
    return;
}

signed main(void){

    ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    cin >> l >> r;

    Prime(ceil(sqrt(r)));

    for(int i = l ; i <= r ; ++i){
        phi[i - l + 1] = i;
        rest[i - l + 1] = i;
    }

    for(int i = 1 ; i <= cnt ; ++i){
        int p = prime[i];
        for(int j = (l + p - 1) / p * p ; j <= r ; j += p){
            phi[j - l + 1] = phi[j - l + 1] / p * (p - 1);
            while(rest[j - l + 1] % p == 0)
                rest[j - l + 1] /= p;
        }
    }

    int res = 0;

    for(int i = l ; i <= r ; ++i){
        if(rest[i - l + 1] > 1)phi[i - l + 1] = phi[i - l + 1] / rest[i - l + 1] * (rest[i - l + 1] - 1);
        res = (res + i - phi[i - l + 1]) % mod;
    }

    cout << res << endl;


    return 0;
}
