多米诺骨牌题解

2021-11-12 5 min read # 背包 # 动态规划

最初思路是设 fi,a,bf_{i , a , b}fi,a,b​ 表示所有只考虑前 iii 个骨牌，凑出的 s1s1s1、s2s2s2 值分别为 aaa和 bbb 的方案中的最小次数

但是无论是时间还是空间复杂度都太大，考虑优化

我们发现，不管怎么翻，骨牌的总和是不变的，因此考虑先求出 sumsumsum 即所有骨牌点数总和，然后考虑固定一个 aaa ，那么 bbb 我们是可以用 sum−asum - asum−a 求出来的。

因此，设 fi,sf_{i , s}fi,s​ 表示所有只考虑前 iii 个骨牌，凑出来 s1s1s1 的值为 sss 的方案中的最小值

但是转移中，我把 fi−1,s−bif_{i - 1 , s - b_i}fi−1,s−bi​​ 写成了 fi−1,s+ai−bif_{i - 1 , s + a_i - b_i}fi−1,s+ai​−bi​​ 一开始还很懊悔自己这么简单的东西都看了题解，但是想了想，自己平时就是不注意，因为我的状态中显示的是只考虑前 iii 个骨牌，因此当 iii 逐渐增大的时候，其实就是逐渐添加骨牌的过程，而 s+ai−bis + a_i - b_is+ai​−bi​ 显示的其实是原来的值发生改变的过程，这两者是不一样的

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 1010;
const int M = 12000;

int n , f[N][M] , a[N] , b[N] , sum , s1 , minn = (1 << 30) , ans = (1 << 30);

int main(void){

    scanf("%d" , &n);

    for(int i = 1 ; i <= n ; ++ i){
        scanf("%d%d" , &a[i] , &b[i]);
        sum += a[i] + b[i];
    }

    memset(f , 0x3f , sizeof(f));

    f[0][0] = 0;

    for(int i = 1 ; i <= n ; ++ i){
        for(int j = sum ; j >= a[i] ; -- j){
            f[i][j] = f[i - 1][j - a[i]];
        }
        for(int j = sum ; j >= b[i] ; -- j){
            f[i][j] = min(f[i][j] , f[i - 1][j - b[i]] + 1);
        }
    }

    int INF = f[n + 1][sum + 1];

    for(int i = 1 ; i <= sum ; ++ i){
        if(f[n][i] != INF){
            minn = min(abs(sum - i - i) , minn);
        }
    }

    for(int i = 1 ; i <= sum ; ++ i){
        if(abs(sum - i - i) == minn){
            ans = min(ans , f[n][i]);
        }
    }

    printf("%d" , ans);

    return 0;

}

由于空间开得比较大，考虑使用滚动数组，其实挺套路的，注意一下每一次都要把 f[i & 1] 初始化为正无穷

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 1010;
const int M = 12000;

int n , f[2][M] , a[N] , b[N] , sum , s1 , minn = (1 << 30) , ans = (1 << 30);

int main(void){

    scanf("%d" , &n);

    for(int i = 1 ; i <= n ; ++ i){
        scanf("%d%d" , &a[i] , &b[i]);
        sum += a[i] + b[i];
    }

    memset(f , 0x3f , sizeof(f));

    f[0][0] = 0;

    for(int i = 1 ; i <= n ; ++ i){
        memset(f[i & 1] , 0x3f , sizeof(f[i & 1]));
        for(int j = sum ; j >= a[i] ; -- j){
            f[i & 1][j] = f[i - 1 & 1][j - a[i]];
        }
        for(int j = sum ; j >= b[i] ; -- j){
            f[i & 1][j] = min(f[i & 1][j] , f[i - 1 & 1][j - b[i]] + 1);
        }
    }

    int INF = f[n + 1 & 1][sum + 1];

    for(int i = 1 ; i <= sum ; ++ i){
        if(f[n & 1][i] != INF){
            minn = min(abs(sum - i - i) , minn);
        }
    }

    for(int i = 1 ; i <= sum ; ++ i){
        if(abs(sum - i - i) == minn){
            ans = min(ans , f[n & 1][i]);
        }
    }

    printf("%d" , ans);

    return 0;

}
