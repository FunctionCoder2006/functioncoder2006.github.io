费解的开关

Posted on 2022-01-01，3 min read

看到这道题，首先联想的是树上的灯那道题

因此我们很容易得到两个性质：

  • 点灯顺序不影响最终结果
  • 每一个灯最多点 111 下

因为每点一下灯，就是对周围的数进行异或运算，显然与顺序无关，最终都是异或那么多次。考虑异或的一个性质，异或两次等于什么都没有做或者说是 异或的奇偶性。因此每一个灯最多点一下

因此，既然顺序不影响最终结果，那么我们不妨一行一行地考虑

然后第三个性质是非常妙的

我们发现，如果我们固定住第一行的状态并且要让所有的灯点亮，我们最多只有一种方案

因为我们考虑，如果这一行进行点击后并没有全部变成 111，那么只能由下一行的点击来点亮。又因为一个灯的点击只影响上下左右和它自己。因此，我们如果想点亮这个灯的话，只能由它的正下方的灯来点击

#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>

using namespace std;

int q , ans , h[10000] , a[10][10] , s[10][10];
char str[10][10];

inline int lowbit(int x){
    return x & (-x);
}

inline int solve(int state){
    int cnt = 0;
    while(state){
        int x = h[lowbit(state)] + 1;
        s[2][x] ^= 1 , s[1][x - 1] ^= 1 , s[1][x + 1] ^= 1 , s[1][x] ^= 1;
        state -= lowbit(state);
        ++ cnt;
    }
    for(int i = 2 ; i <= 5 ; ++ i){
        for(int j = 1 ; j <= 5 ; ++ j){
            if(s[i - 1][j] == 0){
                s[i][j] ^= 1 , s[i - 1][j] ^= 1 , s[i + 1][j] ^= 1 , s[i][j - 1] ^= 1 , s[i][j + 1] ^= 1;
                ++ cnt;
            }
        }
    }
    for(int j = 1 ; j <= 5 ; ++ j){
        if(s[5][j] == 0)
            return (1 << 30);
    }
    return cnt;
}

int main(void){

    scanf("%d" , &q);

    for(int i = 0 ; i <= 6 ; ++ i){
        h[1 << i] = i;
    }

    while(q -- ){
        ans = (1 << 30);
        for(int i = 1 ; i <= 5 ; ++ i){
            scanf("%s" , str[i] + 1);
            for(int j = 1 ; j <= 5 ; ++ j){
                a[i][j] = str[i][j] - '0';
            }
        }
        for(int i = 0 ; i < (1 << 5) ; ++ i){
            memcpy(s , a , sizeof(s));
            ans = min(ans , solve(i));
        }
        if(ans > 6){
            puts("-1");
        }
        else printf("%d\n" , ans);
    }

    return 0;

}
