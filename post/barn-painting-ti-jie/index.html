Barn Painting题解

2021-11-15

因为要求颜色不一样，所以考虑将颜色加入状态

设 f[u][1/2/3]f[u][1/2/3]f[u][1/2/3] 以 uuu 所有以 uuu 为子树，其中 uuu 的颜色分别为 1 2 3 的方案，

以颜色 1 为例，其他两种颜色也是同样的道理，因为 各个子树互不相干，所以，有

fu,1=∏v∈Son(u)(fv,2+fv,3)f_{u , 1} = \prod_{v ∈ Son(u)} (f_{v , 2} + f_{v , 3}) fu,1​=v∈Son(u)∏​(fv,2​+fv,3​)

写法和飞扬的小鸟类似，就是先不管状态合不合法，先去算，最后用 000 覆盖掉不合法的状态

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 1e5 + 10;
const int mod = 1e9 + 7;

int n , k , f[N][5] , colored[N];

int head[N << 1] , ver[N << 1] , nx[N << 1] , idx;
inline void add(int x , int  y){
    ver[idx] = y;
    nx[idx] = head[x] , head[x] = idx ++ ;
    return;
}

inline void dfs(int u , int fa){
    f[u][1] = f[u][2] = f[u][3] = 1;
    for(int i = head[u] ; ~i ; i = nx[i]){
        int v = ver[i];
        if(v == fa)continue;
        dfs(v , u);
        f[u][1] = (1ll * f[u][1] * (f[v][2] + f[v][3])) % mod;
        f[u][2] = (1ll * f[u][2] * (f[v][1] + f[v][3])) % mod;
        f[u][3] = (1ll * f[u][3] * (f[v][1] + f[v][2])) % mod;
    }
    if(colored[u] == 1){
        f[u][2] = f[u][3] = 0;
    }
    if(colored[u] == 2){
        f[u][1] = f[u][3] = 0;
    }
    if(colored[u] == 3){
        f[u][1] = f[u][2] = 0;
    }
    return;
}

int main(void){

    scanf("%d%d" , &n , &k);

    memset(head , -1 , sizeof(head));

    for(int i = 1 ; i < n ; ++ i){
        int x , y;
        scanf("%d%d" , &x , &y);
        add(x , y);
        add(y , x);
    }

    for(int i = 1 ; i <= k ; ++ i){
        int x , y;
        scanf("%d%d" , &x , &y);
        colored[x] = y;
    }

    dfs(1 , -1);

    printf("%d" , ((f[1][1] + f[1][2]) % mod + f[1][3]) % mod);

    return 0;

}
