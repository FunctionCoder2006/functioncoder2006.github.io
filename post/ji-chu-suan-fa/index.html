快速幂

基本思想就是任意一个正整数都可以唯一分解成若干个指数不重复的 222 的次幂之和

因为 ab∗ac=ab+ca ^ b * a ^ c = a ^ {b + c}ab∗ac=ab+c

因此我们考虑将 bbb 二进制分解

ab=ak1∗ak2∗ak3a ^ b = a ^ {k_1} * a ^ {k_2} * a ^ {k_3}ab=ak1​∗ak2​∗ak3​ ..................

其中 kik_iki​ 为 222 的若干次幂

然后我们从低位到高位依次遍历 bbb 的二进制位，可以顺便递推得到 akia ^ {k_i}aki​，因为 a2i=(a2i−1)2a ^ {2 ^ i} = (a ^ {2 ^ {i - 1}}) ^ 2a2i=(a2i−1)2如果发现这个二进制位是 111 的话就累加到答案

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

int a , b , p;

long long power(int a , int b){
    long long ans = 1 , base = a;
    while(b){
        if(b & 1)ans = (ans * base) % p;
        base = base * base % p;
        b >>= 1;
    }
    return ans % p;
}

int main(void){
    
    cin >> a >> b >> p;
    
    cout << power(a , b) << endl;
    
    return 0;
    
}

龟速乘

我们可以考虑使用和快速幂同样的思想，将 bbb 进行二进制拆分

a∗b=a∗(2k1+2k2+2k3+……)a * b = a * (2 ^ {k_1} + 2 ^ {k_2} + 2 ^ {k _ 3} + ……)a∗b=a∗(2k1​+2k2​+2k3​+……)

因为 a∗2i=a∗2i−1∗2a * 2 ^ i = a * 2 ^ {i - 1} * 2a∗2i=a∗2i−1∗2

因此，我们很容易通过遍历 bbb 的二进制位，顺便递推求出每一项

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;
 
long long a , b , p;

long long power(long long a , long long b){
    a %= p , b %= p;
    long long ans = 0 , base = a;
    while(b){
        if(b & 1)ans = (ans + base) % p;
        base = (base + base) % p;
        b >>= 1;
    }
    return ans % p;
}

int main(void){
    
    cin >> a >> b >> p;
    
    cout << power(a , b) << endl;
    
    return 0;
    
}

lowbit运算

使用lowbit遍历二进制位

#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

int n , h[1000000];

inline int lowbit(int x){
    return x & (-x);
}

int main(void){

    cin >> n;

    for(int i = 0 ; i <= 10 ; ++ i){
        h[1 << i] = i;
    }

    while(n){
        cout << h[lowbit(n)] << endl;
        n -= lowbit(n) ;
    }

    return 0;

}
