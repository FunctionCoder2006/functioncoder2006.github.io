看了李煜东的书后感受颇深

但是我个人认为，现在 OIerOIerOIer 写的大多都是进行过滚动数组优化的树形 dpdpdp ，今天想了好一会儿才想到原始的树形 dpdpdp 应该怎么做

首先，这道题可以抽象成是一个分组背包，每一个子树都是若干个互不相干的组，每一组都只能选一个物品，体积（即选多少课程）为 ttt ，价值为 f[v][t]f[v][t]f[v][t]

但是后来我对 f[u][j]=min(f[u][j],f[u][j−t]+f[v][t]f[u][j] = min(f[u][j] , f[u][j - t] + f[v][t]f[u][j]=min(f[u][j],f[u][j−t]+f[v][t] 不太理解，这叫分组背包的写法？

然后我就想，如何转换成分组背包的写法。

考虑设 fu,i,jf_{u , i , j}fu,i,j​ 表示所有以 uuu 为子树，只考虑 uuu 的前 iii 个子树，已经分配了 jjj 个体积，有

fu,i,j=min(fu,i,j,fu,i−1,j−t+fv,subv,t)，t∈[0,min(sizv,j−1)]f_{u , i , j} = min(f_{u , i , j} , f_{u , i - 1 , j - t} + f_{v , sub_v , t})，t∈[0 , min(siz_v , j - 1)] fu,i,j​=min(fu,i,j​,fu,i−1,j−t​+fv,subv​,t​)，t∈[0,min(sizv​,j−1)]

其中，subvsub_vsubv​ 表示 vvv 有多少个子树

然后考虑如何转换成今天所写的写法

我们发现 iii 层决策只和 i−1i - 1i−1 层决策有关，因此考虑使用滚动数组，但是有个问题，这不是递归吗，能用滚动数组吗？可以的，因为递归其实就是确定一个先把子树算好，然后向上传递，把父亲算好
的 dpdpdp 顺序而已

因此，就有了今天的写法

fu,j=min(fu,j,fu,j−t+fv,t)，t∈[0,min(sizv,j−1)]f_{u , j} = min(f_{u , j} , f_{u , j - t} + f_{v , t})，t∈[0 , min(siz_v , j - 1)] fu,j​=min(fu,j​,fu,j−t​+fv,t​)，t∈[0,min(sizv​,j−1)]

这里注意不是 jjj 而是 j−1j - 1j−1 因为我们将 fu,1=vuf_{u , 1} = v_ufu,1​=vu​ 所以我们默认为我们先给 uuu 分配了 1 个体积

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 330;

int n , m , f[N][N];

int head[N << 1] , ver[N << 1] , nx[N << 1] , idx , siz[N];
inline void add(int x , int y){
    ver[idx] = y;
    nx[idx] = head[x] , head[x] = idx ++ ;
    return;
}

inline void dfs(int u , int fa){
    siz[u] = 1;
    for(int i = head[u] ; ~i ; i = nx[i]){
        int v = ver[i];
        if(v == fa)continue;
        dfs(v , u);
        siz[u] += siz[v];
        for(int j = min(m , siz[u]) ; j >= 0 ; -- j){
            for(int k = min(siz[v] , j - 1) ; k >= 0 ; -- k){
                f[u][j] = max(f[u][j] , f[u][j - k] + f[v][k]);
            }
        }
    }
    return ;
}

int main(void){

    scanf("%d%d" , &n , &m);

    int y , z;

    memset(head , -1 , sizeof(head));

    for(int x = 1 ; x <= n ; ++ x){
        scanf("%d%d" , &y , &f[x][1]);
        add(x , y);
        add(y , x);
    }

    m ++ ;

    dfs(0 , -1);

    printf("%d" , f[0][m]);

    return 0;

}

后来遇见了一些稍微有些细节的树形 dpdpdp，这时候才反应过来自己的有些地方没有注意到

这道题是树上分组背包没错，但是题目中要求是恰好选 mmm 门课程，而我们的状态里“已经”也是“恰好”的意思，所以有些状态是非法的，我们需要让 dpdpdp 数组先初始化成负无穷，让那些非法状态更新不到我的合法状态

那么，之前的代码没有将 dpdpdp 数组初始化为负无穷而是 0 ，求出来的是“不超过”mmm 门课程的最大学分，那么，还对不对呢？这道题比较特殊，因为选的课程越多，求出来的最大值肯定越大。所以，对于本题，按照“不超过” mmm 门课程求出来的最大值一定等于“恰好”mmm 门课程求出来的最大值

还有 不要忘记将 f[0][1]f[0][1]f[0][1] 初始化为0，因为这个调了半天

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 330;

int n , m , f[N][N];

int head[N << 1] , ver[N << 1] , nx[N << 1] , idx , siz[N];
inline void add(int x , int y){
    ver[idx] = y;
    nx[idx] = head[x] , head[x] = idx ++ ;
    return;
}

inline void dfs(int u , int fa){
    siz[u] = 1; f[u][0] = 0;
    for(int i = head[u] ; ~i ; i = nx[i]){
        int v = ver[i];
        if(v == fa)continue;
        dfs(v , u);
        siz[u] += siz[v];
        for(int j = min(siz[u] , m) ; j >= 0 ; -- j){
            for(int k = 0 ; k <= min(siz[v] , j - 1) ; ++ k){
                f[u][j] = max(f[u][j] , f[u][j - k] + f[v][k]);
            }
        }
        
    }

    return ;
}

int main(void){

    scanf("%d%d" , &n , &m);

    int y , z;

    memset(head , -1 , sizeof(head));
    memset(f , -0x3f , sizeof(f));

    for(int x = 1 ; x <= n ; ++ x){
        scanf("%d%d" , &y , &f[x][1]);
        add(x , y);
        add(y , x);
    }

    m ++ ;

    f[0][1] = 0;

    dfs(0 , -1);

    printf("%d" , f[0][m]);

    return 0;

}
