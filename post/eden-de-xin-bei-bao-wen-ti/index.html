Eden的新背包问题

2021-11-14 #单调队列 #背包 #动态规划

单调队列优化多重背包，做一遍前缀背包和后缀背包，每一次询问考虑分配体积，合并正反背包

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 1e3 + 10;
const int m = 1000;

int n , w[N] , v[N] , s[N] , q[N] , f[N][N] , g[N][N];
int Q , x , V;

inline int du(void){
    int x = 0 , f = 1;
    char ch = getchar();
    while(ch > '9' || ch < '0'){if(ch == '-')f = -1;ch = getchar();}
    while(ch >= '0' && ch <= '9'){x = x * 10 + ch - '0';ch = getchar();}
    return x * f;
}
inline void write(int x){
    if(x < 0){putchar('-');x = -x;}
    if(x > 9)write(x / 10);
    putchar(x % 10 ^ 48);
    return;
}

inline int solve(int i , int u , int k){
    return f[i - 1][u + k * w[i]] - k * v[i];
}
inline int calc(int i , int u , int k){
    return g[i + 1][u + k * w[i]] - k * v[i];
}

int main(void){

    n = du();

    for(int i = 1 ; i <= n ; ++ i){
        w[i] = du() , v[i] = du() , s[i] = du();
    }

    for(int i = 1 ; i <= n ; ++ i){
        int Len = s[i];
        if(m / w[i] < s[i])Len = m / w[i];  
        for(int u = 0 ; u < w[i] ; ++ u){
            int hh = 1 , tt = 0;
            for(int k = 0 ; k <= (m - u) / w[i] ; ++ k){
                while(hh <= tt && solve(i , u , q[tt]) <= solve(i , u , k))
                    -- tt;
                while(hh <= tt && q[hh] < k - Len)
                    ++ hh;
                q[++ tt] = k;
                f[i][u + k * w[i]] = solve(i , u , q[hh]) + k * v[i];
            }
        }
    }

    for(int i = n ; i >= 1 ; -- i){
        int Len = s[i]; 
        if(m / w[i] < s[i])Len = m / w[i];
        for(int u = 0 ; u < w[i] ; ++ u){
            int hh = 1 , tt = 0;
            for(int k = 0 ; k <= (m - u) / w[i] ; ++ k){
                while(hh <= tt && calc(i , u , q[tt]) <= calc(i , u , k))
                    -- tt;
                while(hh <= tt && q[hh] < k - Len)
                    ++ hh;
                q[++ tt] = k;
                g[i][u + k * w[i]] = calc(i , u , q[hh]) + k * v[i];
            }
        }
    }

    Q = du();

    while(Q -- ){
        x = du() , V = du();
        ++ x;
        int ans = 0;
        for(int k = 0 ; k <= V ; ++ k){
            ans = max(ans , f[x - 1][k] + g[x + 1][V - k]);
        }
        write(ans) , putchar('\n');
    }

    return 0;

    
}
