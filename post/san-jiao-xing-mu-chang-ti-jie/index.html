三角形牧场题解

Posted on 2021-11-14，3 min read

首先想到设 fa,b,c=0/1f_{a , b , c} = 0/1fa,b,c​=0/1 表示三角形三边长度分别为 aaa bbb ccc 能否被凑成

但是这样显然是过不去的，考虑优化

因为所有的棍子都要用到，所以棍子的总长度是不变的，因此如果令 aaa bbb 固定，那么 ccc 就可以求出来即 sum−a−bsum - a - bsum−a−b ，于是，ccc 这一维就可以被压掉

但是时间复杂度还是高，考虑使用三角形不等式即 a+b<ca + b < ca+b<c 所有的边都不能超过总长度的一半，因为如果超了的话是铁定凑不成的

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 810;

int n , a[N] , sum;
bool f[N][N];

double solve(int a , int b , int c){
    if(a + b <= c || a + c <= b || b + c <= a)
        return 0.0;
    double p = (a + b + c) * 1.0 / 2;
    return sqrt(p * (p - a) * (p - b) * (p - c));
}

int main(void){

    scanf("%d" , &n);

    for(int i = 1 ; i <= n ; ++ i){
        scanf("%d" , &a[i]);
        sum += a[i];
    }

    f[0][0] = true;

    for(int i = 1 ; i <= n ; ++ i){
        for(int j = 800 ; j >= 0 ; -- j){
            for(int k = 800 ; k >= 0 ; -- k){
                if(j >= a[i])f[j][k] |= f[j - a[i]][k];
                if(k >= a[i])f[j][k] |= f[j][k - a[i]];
            }
        }
    }

    double ans = 0.0;

    for(int a = 0 ; a <= 800 ; ++ a){
        for(int b = 0 ; b <= 800 ; ++ b){
            if(f[a][b])ans = max(ans , solve(a , b , sum - a - b));
        }
    }

    if(ans == 0.0){
        puts("-1");
        return 0;
    }

    //cout << sum << endl;

    printf("%d" , (int)(ans * 100));

    return 0;

}

这道题的 dpdpdp 数组并不是直接用来求答案的，而是利用状态辅助计算
