二叉苹果树

2021-11-15 #树形dp #背包 #动态规划

这道题当时磨蹭了半天原因是被一些特判（t=0/j=0t = 0/j = 0t=0/j=0）绊住了，后来发现读错题了……

其实和选课也差不多，有些细节还是需要稍加注意

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 550;

int n , m , siz[N] , f[N][N];

int head[N << 1] , nx[N << 1] , ver[N << 1] , w[N << 1] , idx;
inline void add(int x , int y , int z){
    ver[idx] = y , w[idx] = z;
    nx[idx] = head[x] , head[x] = idx ++ ;
    return;
}

inline void dfs(int u , int fa){
    siz[u] = 1 , f[u][0] = 0;
    for(int i = head[u] ; ~i ; i = nx[i]){
        int v = ver[i];
        if(v == fa)continue;
        dfs(v , u);
        siz[u] += siz[v];
        for(int j = min(m , siz[u] - 1) ; j >= 1 ; -- j){
            for(int t = min(j , siz[v]) ; t >= 1 ; -- t){
                f[u][j] = max(f[u][j] , f[u][j - t] + f[v][t - 1] + w[i]);
            }
        }
    }

    return;

}

int main(void){
    
    scanf("%d%d" , &n , &m);
    
    memset(head , -1 , sizeof(head));

 //   memset(f , ~0x3f , sizeof(f));

    for(int i = 1 ; i < n ; ++ i){
        int x , y , z;
        scanf("%d%d%d" , &x , &y , &z);
        add(x , y , z);
        add(y , x , z);
    }

    dfs(1 , -1);

    printf("%d" , f[1][m]);

    return 0;

}
