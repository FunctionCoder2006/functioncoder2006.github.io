联合权值题解

2021-11-26 #前缀和 #树形dp #动态规划

一道不错的题

考虑到两个距离为 222 的点 uuu vvv要么互为兄弟节点，要么是其中的一个点 uuu 向下走 222 步到达点 vvv

然后我们考虑固定一个点 uuu ，求出与 tatata 相距为 222 的点的权值和

其实也很简单，让 uuu 固定不动，设 sus_usu​ 表示 uuu 的子节点的权值和，计算贡献就可以了

在处理兄弟节点的贡献时，因为那种处理方式一定覆盖了最优解，所以是正确的

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 2e5 + 10;
const int mod = 10007;

int n , w[N] , f[N];
long long s[N] , ans , res;

int head[N << 1] , nx[N << 1] , ver[N << 1] , idx;
inline void add(int x , int y){
    ver[idx] = y;
    nx[idx] = head[x] , head[x] = idx ++ ;
    return;
}

inline void dfs(int u , int fa){
    for(int i = head[u] ; ~i ; i = nx[i]){
        int v = ver[i];
        if(v == fa)continue;
        dfs(v , u);
        ans += 1ll * w[u] * s[v]; ans %= mod;
        ans += 1ll * w[v] * s[u]; ans %= mod;
        s[u] += w[v]; s[u] %= mod;
        res = max(res , 1ll * w[u] * f[v]);
        res = max(res , 1ll * w[v] * f[u]);
        f[u] = max(f[u] , w[v]);
    }
    return;
}

int main(void){

    scanf("%d" , &n);

    memset(head , -1 , sizeof(head));

    for(int i = 1 ; i < n ; ++ i){
        int x , y;
        scanf("%d%d" , &x , &y);
        add(x , y);
        add(y , x);
    }

    for(int i = 1 ; i <= n ; ++ i){
        scanf("%d" , &w[i]);
    }

    dfs(1 , -1);

    printf("%lld %lld" , res , ans * 2 % mod);
    
    return 0;
}
