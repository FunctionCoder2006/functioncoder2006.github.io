分级

Posted on 2021-12-18，2 min read

这道题可以说是一个套路题或者结论题吧

结论是：最优解中的 bbb 序列一定全都在 aaa 中出现过，这个 yxcyxcyxc 给出了非常好的证明。这里不再赘述

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 2100;

int n , a[N] , c[N] , f[N][N]; //c[i]表示a[i]排序后的数组

inline int get_up(void){
    memset(f , 0x3f , sizeof(f));
    int ans = (1 << 30);
    for(int k = 1 ; k <= n ; ++ k){
        f[0][k] = 0;
    }
    for(int i = 1 ; i <= n ; ++ i){
        int maxn = (1 << 30);
        for(int j = 1 ; j <= n ; ++ j){
            maxn = min(maxn , f[i - 1][j]);
            f[i][j] = min(f[i][j] , maxn + abs(c[j] - a[i]));
        }
    }
    for(int i = 1 ; i <= n ; ++ i){
        ans = min(ans , f[n][i]);
    }
    return ans;
}
inline int get_down(void){
    memset(f , 0x3f , sizeof(f));
    for(int k = 1 ; k <= n ; ++ k){
        f[0][k] = 0;
    }
    int ans = (1 << 30);
    for(int i = 1 ; i <= n ; ++ i){
        int maxn = (1 << 30);
        for(int j = n ; j >= 1 ; -- j){
            maxn = min(maxn , f[i - 1][j]);
            f[i][j] = min(f[i][j] , maxn + abs(c[j] - a[i]));
        }
    }
    for(int i = 1 ; i <= n ; ++ i){
        ans = min(ans , f[n][i]);
    }
    return ans;
}

int main(void){

    cin >> n;

    for(int i = 1 ; i <= n ; ++ i){
        cin >> a[i];
        c[i] = a[i];
    }

    sort(c + 1 , c + 1 + n);

    cout << min(get_up() , get_down()) << endl;

    return 0;

}
