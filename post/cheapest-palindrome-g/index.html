Cheapest Palindrome G

Posted on 2021-10-19，3 min read

这道题其实不算很难，主要是来记录一下坑点

一开始，我将长度和 1 和长度为 2 的 dpdpdp 数组全都初始化了，然后区间 dpdpdp 时直接从 Len=3Len = 3Len=3 开始枚举了，但是我们会发现，Len=2Len = 2Len=2同样可以进行删除和插入操作

然后提交了之后是 90 ptsptspts

又发现，当 Len=2Len = 2Len=2 时，如果此时是一个回文，而我将其赋值成了 fl+1,r−1f_{l + 1 , r - 1}fl+1,r−1​ 这样就会导致 此时的 dpdpdp 值为正无穷。

所以当 Len=2Len = 2Len=2 还需要再特判一下

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 2010;

int n , m , del[N] , Ins[N];
int f[N][N];
char str[N] , opt[3];

int main(void){

 scanf("%d%d" , &n , &m);

 scanf("%s" , str + 1);

 for(int i = 1 ; i <= n ; ++ i){
     int a , b;
     char c;
     scanf("%s%d%d" , opt , &a , &b);
     c = opt[0] - 'a' + 1;
     Ins[c] = a , del[c] = b;
 }

 memset(f , 0x3f , sizeof(f));

 for(int i = 1 ; i <= m ; ++ i){
     f[i][i] = 0;
 }

 for(int Len = 2 ; Len <= m ; ++ Len){
     for(int l = 1 ; l + Len - 1 <= m ; ++ l){
         int r = l + Len - 1;
         char L = str[l] - 'a' + 1 , R = str[r] - 'a' + 1;
         if(L == R){
             if(Len == 2){
                 f[l][r] = 0;
                 continue;
             }
             else f[l][r] = f[l + 1][r - 1];
         }
         f[l][r] = min(f[l][r] , f[l][r - 1] + del[R]);
         f[l][r] = min(f[l][r] , f[l + 1][r] + del[L]);
         f[l][r] = min(f[l][r] , f[l][r - 1] + Ins[R]);
         f[l][r] = min(f[l][r] , f[l + 1][r] + Ins[L]);
     }
 }

 printf("%d" , f[1][m]);

 return 0;

}
