廊桥分配题解

Posted on 2021-10-27，3 min read

暴力的思路应该很好想，如果给国内机场分配 kkk 个，那么国际机场就要分配 n−kn - kn−k 个，然后分别用堆模拟，算出一共可以停多少飞机。在枚举 kkk 的时候打一遍擂台求出最小值。

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<queue>

using namespace std;

const int N = 1e5 + 10;

int n , m1 , m2 , ans;

inline int du(void){
	int x = 0 , f = 1;
	char ch = getchar();
	while(ch > '9' || ch < '0' ){if(ch == '-')f = -1;ch = getchar();}
	while(ch >= '0' && ch <= '9'){x = x * 10 + ch - '0';ch = getchar();}
	return x * f;
}
inline void write(int x){
	if(x < 0){putchar('-');x = -x;}
	if(x > 9)write(x / 10);
	putchar(x % 10 ^ 48);
	return;
}

struct node{
	int l , r;
}a[N] , b[N];

bool cmp(node x , node y){
	return x.l < y.l;
}

inline int solve_a(int k){
	
	priority_queue<int , vector<int> , greater<int> > que;
	
	int cnt = 0;
	
	for(int i = 1 ; i <= m1 ; ++ i){
		while(que.size() && que.top() <= a[i].l)
			que.pop();
		if(que.size() < k){
			que.push(a[i].r);
			++ cnt;
		}	
	}
	
	return cnt;
}

inline int solve_b(int k){
	
	priority_queue<int , vector<int> , greater<int> > que;
	
	int cnt = 0;
	
	for(int i = 1 ; i <= m2 ; ++ i){
		while(que.size() && que.top() <= b[i].l)
			que.pop();
		if(que.size() < k){
			que.push(b[i].r);
			++ cnt;
		}
	}
	
	return cnt;
	
}

int main(void){
	
//	freopen("airport.in" , "r" , stdin);
//	freopen("airport.out" , "w" , stdout);
	
	n = du() , m1 = du() , m2 = du();
	
	for(int i = 1 ; i <= m1 ; ++ i){
		a[i].l = du() , a[i].r = du();
	}
	
	sort(a + 1 , a + 1 + m1 , cmp);
	
	for(int i = 1 ; i <= m2 ; ++ i){
		b[i].l = du() , b[i].r = du();
	}
	
	sort(b + 1 , b + 1 + m2 , cmp);
	
	for(int k = 1 ; k <= n ; ++ k){
		ans = max(ans , solve_a(k) + solve_b(n - k));
	}
	
	write(ans);
	
	return 0;
	
}

这样复杂度是 O(N2logN)O(N^2log_N)O(N2logN​)，复杂度过高。

枚举 kkk 的时候都要模拟一遍有些浪费，因此考虑预处理，是的枚举 kkk 的时候直接用。

以国内机场举例，国际机场也是同样的道理。

假设国内机场有无穷多个廊桥，我们模拟一遍就可以知道国内机场所有飞机会停到那一个廊桥，进而知道所有有飞机的廊桥先后一共停了多少飞机。
