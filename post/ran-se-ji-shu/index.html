#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 5010;
const int mod = 1e9 + 7;

inline int du(void){
    int x = 0 , f = 1;
    char ch = getchar();
    while(ch > '9' || ch < '0'){if(ch == '-')f = -1;ch = getchar();}
    while(ch >= '0' && ch <= '9'){x = x * 10 + ch - '0';ch = getchar();}
    return x * f;
}
inline void write(int x){
    if(x < 0){putchar('-');x = -x;}
    if(x > 9)write(x / 10);
    putchar(x % 10 ^ 48);
    return;
}

int n , m , f[N][N] , sum[N];

int head[N << 1] , ver[N << 1] , nx[N << 1] , idx;
inline void add(int x , int y){
    ver[idx] = y;
    nx[idx] = head[x] , head[x] = idx ++ ;
    return;
}

inline void dfs(int u , int fa){
    for(int i = head[u] ; ~i ; i = nx[i]){
        int v = ver[i];
        if(v == fa)continue;
        dfs(v , u);
        for(int j = 1 ; j <= m ; ++ j){
            f[u][j] = (1ll * f[u][j] * ((sum[v] - f[v][j] + mod) % mod)) % mod;
        }
    }
    for(int i = 1 ; i <= m ; ++ i){
        sum[u] = (sum[u] + f[u][i]) % mod;
    }
    return; 
}

int main(void){

    n = du() , m = du();

    for(int i = 1 ; i <= n ; ++ i){
        int k , c;
        k = du();
        for(int j = 1 ; j <= k ; ++ j){
            c = du();
            f[i][c] = 1;
        }
    }   

    memset(head , -1 , sizeof(head));

    for(int i = 1 ; i < n ; ++ i){
        int x , y;
        x = du() , y = du();
        add(x , y);
        add(y , x);
    }

    dfs(1 , -1);

    write(sum[1] % mod);

    return 0;
}
