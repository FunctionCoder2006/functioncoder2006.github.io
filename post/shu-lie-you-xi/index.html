数列游戏题解

2021-10-19 4 min read # 区间DP # 动态规划

介绍一种比较暴力的做法。

跟其他题解很想，设 gl,rg_{l , r}gl,r​ 表示区间 [L,R][L , R][L,R] 能否完全消除。

我们很容易得到：

gl,r=gl+1,r−1g_{l , r} = g_{l + 1 , r - 1}gl,r​=gl+1,r−1​ & (al,ar)=1(a_l , a_r) = 1(al​,ar​)=1

gl,rg_{l , r}gl,r​ |= gl,kg_{l , k}gl,k​ & gk+1,rg_{k + 1 , r}gk+1,r​其中k∈[l,r]k∈[l , r]k∈[l,r]

其实就是两种方式，一种是消除 [l+1,r−1][l + 1 , r - 1][l+1,r−1] 的区间最后消除 ala_lal​ 和 ara_rar​ ；另一种是将 [l,r][l , r][l,r] 分成两个区间两个区间然后分别消除。

设 fl,rf_{l , r}fl,r​ 表示对 [l,r][l , r][l,r] 进行操作（不要求完全消除）的最大收益

如果 [l,r][l , r][l,r] 可以被完全消除，则 fl,rf_{l , r}fl,r​ 就是 ∑i=lnbi\sum_{i = l}^{n} b_i∑i=ln​bi​

否则考虑讲一个区间拆成两个区间，然后合并，即 fl,r=max(fl,r,fl,k+fk+1,r)f_{l , r} = max(f_{l , r} , f_{l , k} + f_{k + 1 , r})fl,r​=max(fl,r​,fl,k​+fk+1,r​)

有的人可能会问，两个区间合并的时候接壤的部分你没有考虑到呀……

是这样的，虽然在当前我在考虑我的两个区间 [l,k][l , k][l,k] 和 [k+1,r][k + 1 , r][k+1,r] 时没有考虑接壤部分，但是当我的 kkk 继续增大直到 [l,r][l , r][l,r] 覆盖掉两个区间的接壤部分时，我们认为 [l,k][l , k][l,k] 是完全处理好的，即接壤部分我们已经考虑了，所以就不需要担心两个区间接壤的问题。

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 810;

int n , a[N] , b[N];
long long f[N][N] , sum[N];
bool g[N][N];

inline int gcd(int a , int b){
    if(b == 0)return a;
    return gcd(b , a % b);
}

int main(void){

    scanf("%d" , &n);

    for(int i = 1 ; i <= n ; ++ i){
        scanf("%d" , &a[i]);
    }
    for(int i = 1 ; i <= n ; ++ i){
        scanf("%d" , &b[i]);
        sum[i] = sum[i - 1] + b[i];
    }

    for(int i = 2 ; i <= n ; ++ i){
        if(gcd(a[i] , a[i - 1]) > 1)
            f[i - 1][i] = b[i - 1] + b[i] , g[i - 1][i] = true;
        else f[i - 1][i] = 0;
    }

    for(int Len = 3 ; Len <= n ; ++ Len){
        for(int l = 1 ; l + Len - 1 <= n ; ++ l){
            int r = l + Len - 1;
            if(g[l + 1][r - 1] && gcd(a[l] , a[r]) > 1)
                g[l][r] = true;
            for(int k = l ; k < r ; ++ k){
                g[l][r] |= (g[l][k] & g[k + 1][r]);
            }
        }
    }

    for(int Len = 3 ; Len <= n ; ++ Len){
        for(int l = 1 ; l + Len - 1 <= n ; ++ l){
            int r = l + Len - 1;
            if(g[l][r]){
                f[l][r] = sum[r] - sum[l - 1];
                continue;
            }
            for(int k = l ; k < r ; ++ k){
                f[l][r] = max(f[l][r] , f[l][k] + f[k + 1][r]);
            }
        }
    }

    printf("%lld" , f[1][n]);

    return 0;

}
