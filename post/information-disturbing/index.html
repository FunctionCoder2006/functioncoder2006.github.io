Information Disturbing

Posted on 2021-11-30，4 min read

弥补了一维树形 dpdpdp 的遗憾

设 fuf_ufu​ 表示以 uuu 为子树进行切割的最小花费，其中我们需要考虑 fuf_ufu​ 是否表示让其所有叶节点不能到达根节点

我们发现，如果我们选择切割这条边，那么子树不需要进行任何切割，因为切了也是白切

重点是考虑不去切割这条边。我们假设 fuf_{u}fu​ 状态里有让其所有叶节点不能到达根节点的含义，那么如果我们想要符合题设，我们就必须保证以 vvv 为子树恰好所有叶节点不能到达根节点，我们发现 vvv 的要求是符合我们的状态的定义的，因此状态里面是有这个含义的。因此，有

fu=∑i=1nmin(fv,wu,v)f_u = \sum_{i = 1}^{n} min(f_{v} , w_{u , v}) fu​=i=1∑n​min(fv​,wu,v​)

其次就是二分答案，因为最大值越大，符合要求的边数越多；边数越多，就越灵活，方案越多；而这最大值更大的方案一定覆盖了最大值比较小的方案，所以，求出来的最小花费就越小，越有可能满足题设的要求。

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 1e3 + 10;

int n , m , f[N] , son[N];

int head[N << 1] , nx[N << 1] , ver[N << 1] , w[N << 1] , idx;
inline void add(int x , int y , int z){
    ver[idx] = y , w[idx] = z;
    nx[idx] = head[x] , head[x] = idx ++ ;
    return;
}

inline void dfs(int u , int fa , int control){
    for(int i = head[u] ; ~i ; i = nx[i]){
        int v = ver[i];
        if(v == fa)continue;
        dfs(v , u , control);
        ++ son[u];
    }
    if(son[u] > 0)f[u] = 0;
    for(int i = head[u] ; ~i ; i = nx[i]){
        int v = ver[i];
        if(v == fa)continue;
        f[u] += min((w[i] <= control ? w[i] : (1 << 20)) , f[v]);
    }
    return;
}

int main(void){

    while(1){        

        scanf("%d%d" , &n , &m);

        if(n == 0 && m == 0)break;

        int x , y , z;

        memset(head , -1 , sizeof(head));
        idx = 0;

        for(int i = 1 ; i < n ; ++ i){
            scanf("%d%d%d" , &x , &y , &z);
            add(x , y , z);
            add(y , x , z);
        }

        int L = 0 , R = 1e6 , ans = -1;

        while(L <= R){
            int mid = (L + R) >> 1;
            for(int i = 1 ; i <= n ; ++ i){
                f[i] = (1 << 20);
            }
            memset(son , 0 , sizeof(son));
            dfs(1 , -1 , mid);
            if(f[1] <= m){
                R = mid - 1;
                ans = mid;
            }
            else L = mid + 1;
        }

        printf("%d\n" , ans);

    }

    return 0;

}

注意不要忘记给 sonsonson 数组初始化为 000；dpdpdp 数组也不是初始化为 0x3f0x3f0x3f 就完事儿了，只有叶节点才是 0x3f0x3f0x3f
