二进制拆分优化多重背包

2021-10-29 2 min read # 背包 # 动态规划

其实挺简单的，就是需要注意两个点，一个是不要忘了特判 si>0s_i > 0si​>0 ，另一个是需要 −−s[i]--s[i]−−s[i]因为输入的时候已经默认为装入了 k=1k = 1k=1 的 w[i]w[i]w[i] 和 v[i]v[i]v[i]

#include<bits/stdc++.h>

using namespace std;

const int N = 2e4 + 10;

int n , m , w[N] , v[N] , f[N] , s[N];

int main(void){

    ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    cin >> n >> m;

    for(int i = 1 ; i <= n ; ++ i){
        cin >> w[i] >> v[i] >> s[i];
    }

    int cnt = n;
 
    for(int i = 1 ; i <= n ; ++ i){
        -- s[i];
        for(int k = 2 ; k <= s[i] ; k <<= 1){
            w[++ cnt] = w[i] * k , v[cnt] = v[i] * k;
            s[i] -= k;
        }
        if(s[i] > 0){
            w[++ cnt] = s[i] * w[i] , v[cnt] = s[i] * v[i];
        }
    }

    for(int i = 1 ; i <= cnt ; ++ i)
        for(int j = m ; j >= w[i] ; -- j)
            f[j] = max(f[j] , f[j - w[i]] + v[i]);
    
    cout << f[m] << endl;

    return 0;

}
