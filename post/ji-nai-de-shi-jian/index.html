挤奶的时间

Posted on 2021-12-09，4 min read

不知道自己在干什么，一晚上一直重复地写 fi=max(fi,fj+wi)f_i = max(f_i , f_j + w_i)fi​=max(fi​,fj​+wi​)

回归正题

设 fif_ifi​ 表示所有结束时间为 iii 的方案中得到的最大值，每一次可以选择挤奶也可以选择不挤奶，因此，有

fi=max(fi−1,fsti−R+wi)f_i = max(f_{i - 1} , f_{st_i - R} + w_i) fi​=max(fi−1​,fsti​−R​+wi​)

我们换一种思考方式，给 fif_ifi​ 的含义添加含义，令 fif_ifi​ 表示所有选择结束时间为 iii 的挤奶时间的方案中的最大值，因此，有

fi=max(fi,fj+wi),j∈[1,sti−R]f_{i} = max(f_i , f_j + w_i) , j ∈[1 , st_i - R] fi​=max(fi​,fj​+wi​),j∈[1,sti​−R]

然后我们把 wiw_iwi​ 提出来，目标就转化成了求最小的 fjf_jfj​ 。其实也很简单，我们只需要开一个前缀 maxmaxmax 数组就可以了，设 gig_igi​ 表示前 iii 个 fif_ifi​ 的最大值，因此，原式可化为

fi=gsti−R+wif_i = g_{st_i - R} + w_i fi​=gsti​−R​+wi​

注意，这里不需要考虑不挤奶，因为状态含义就强制我们必须挤奶，要是不能挤奶的话就记为 000

然后考虑求 ggg ，有

gi=max(gi−1,fi)g_i = max(g_{i - 1} , f_i) gi​=max(gi−1​,fi​)

然后我们将 gsti−R+wig_{st_i - R} + w_igsti​−R​+wi​ 代换 fif_ifi​，因此，有

gi=max(gi−1,gsti−R+wi)g_i = max(g_{i - 1} , g_{st_i - R} + w_i) gi​=max(gi−1​,gsti​−R​+wi​)

我们发现我们得到了和上一种方法一样的式子。

做了一些题，发现 添加状态、前缀优化、代换 一直在用

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<vector>

using namespace std;

const int N = 1e6 + 10;

int n , m , d , g[N];

inline int du(void){
    int x = 0 , f = 1;
    char ch = getchar();
    while(ch > '9' || ch < '0'){if(ch == '-')f = -1;ch = getchar();}
    while(ch >= '0' && ch <= '9'){x = x * 10 + ch - '0';ch = getchar();}
    return x * f;
}
inline void write(int x){
    if(x < 0){putchar('-');x = -x;}    
    if(x > 9)write(x / 10);
    putchar(x % 10 ^ 48);
    return;
}

struct node{
    int st , v;
    node(){}
    node(int _st , int _v){
        st = _st , v = _v;
    }
};

vector<vector<node> > vec(N); 

int main(void){

    n = du() , m = du() , d = du();

    for(int i = 1 ; i <= m ; ++ i){
        int st , ed , v;
        st = du() , ed = du() , v = du();
        vec[ed].push_back(node(st , v));
    }

    for(int i = 1 ; i <= n ; ++ i){
        g[i] = g[i - 1];
        for(int j = 0 ; j < vec[i].size() ; ++ j){
            g[i] = max(g[i] , g[max(0 , vec[i][j].st - d)] + vec[i][j].v);//
        }
    }

    write(g[n]);

    return 0;

}
