这道题有两种做法

因为选出来的序列要保持公差一定，所以，我们考虑枚举公差，将公差一定

对于每一个公差 ttt，设 fif_ifi​ 表示所有以 iii 为结尾的等差数列的方案，因此，有

fi=∑i=1i−1fj，aj=ai−tf_i = \sum_{i = 1}^{i - 1} f_j ，a_j = a_i - t fi​=i=1∑i−1​fj​，aj​=ai​−t

我们发现可以使用前缀和优化，设 cnti,xcnt_{i , x}cnti,x​ 表示所有 j∈[1,i]j ∈[1 , i]j∈[1,i] 的 fjf_{j}fj​ 之和并且满足 aj=xa_j = xaj​=x

因此，有 fi=cnti−1,ai−t+1f_i = cnt_{i - 1 , a_i - t} + 1fi​=cnti−1,ai​−t​+1

如果我们将 cntcntcnt 数组压掉一维的话，我们需要先让 fi=cntai−t+1f_i = cnt_{a_i - t} + 1fi​=cntai​−t​+1，然后将 fif_ifi​ 累加到 cntaicnt_{a_i}cntai​​ 上，保证转移 fif_ifi​ 的时候使用的是前 i−1i - 1i−1 的信息

需要注意的是 cntcntcnt 数组长度不要开得太大了以免 TLETLETLE

接下来是第二种做法

我们很快发现我们枚举的很多公差是没有用的

如果我们确定了最后一个数和倒数第二个数，那么我们就确定了等差数列的公差。因此，考虑枚举最后两个数是多少

设 fi,jf_{i , j}fi,j​ 表示所有最后一个数是 iii ，倒数第二个数是 jjj 的等差数列的方案，因此，有

fi,j=∑k=1j−1fj,k，aj−ak=ai−ajf_{i , j} = \sum_{k = 1}^{j - 1} f_{j , k}，a_j - a_k = a_i - a_j fi,j​=k=1∑j−1​fj,k​，aj​−ak​=ai​−aj​

依然使用前缀和可以将复杂度优化到 O(n2)O(n^2)O(n2)

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>
#define int long long 

using namespace std;

const int N = 5e4 + 10;
const int mod = 998244353;

int n , a[N] , maxt;
long long f[N] , cnt[N] , ans;

signed main(void){

    scanf("%lld" , &n);
    
    for(int i = 1 ; i <= n ; ++ i){
        scanf("%lld" , &a[i]);
        maxt = max(maxt , a[i]);
    }

    for(int t = -maxt ; t <= maxt ; ++ t){
        for(int i = 1 ; i <= n ; ++ i){
            f[i] = 1;
        }
        memset(cnt , 0 , sizeof(cnt));
        cnt[a[1]] = 1;
        for(int i = 2 ; i <= n ; ++ i){
            if(a[i] - t >= 0){
                f[i] += cnt[a[i] - t]; f[i] %= mod;
            }
            cnt[a[i]] += f[i]; cnt[a[i]] %= mod;
            ans += (f[i] - 1); ans %= mod;
        }
    }

    ans += n;

    ans %= mod;
    
    printf("%lld" , ans);

    return 0;

}


#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 1e3 + 10;
const int M = 5e4 + 10;
const int mod = 998244353;

int n , a[N] , f[N][N] , cnt[N][M] , ans;

//cnt[i][j] : k E [1 , i] f[i][k] , a[k] = j

int main(void){
    
    scanf("%d" , &n);

    for(int i = 1 ; i <= n ; ++ i){
        scanf("%d" , &a[i]);
    }

    for(int i = 1 ; i <= n ; ++ i){
        for(int j = 1 ; j < i ; ++ j){
            f[i][j] = 1;
            if(a[j] * 2 - a[i] >= 0)
                f[i][j] += cnt[j][a[j] * 2 - a[i]] , f[i][j] %= mod;
            cnt[i][a[j]] += f[i][j] , cnt[i][a[j]] %= mod;
        }
    }

    for(int i = 1 ; i <= n ; ++ i)
        for(int j = 1 ; j < i ; ++ j)
            ans += f[i][j] , ans %= mod;

    ans += n , ans %= mod;

    printf("%d" , ans % mod);

    return 0;
}
