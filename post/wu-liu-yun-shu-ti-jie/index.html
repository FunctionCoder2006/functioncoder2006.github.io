物流运输题解

2021-12-08

首先我们考虑，在我们求解的过程中，一定是一个连续几天共用一个路线，因此我们考虑区间划分 dpdpdp

设 fif_ifi​ 表示所有将前 iii 天划分的方案中最小花费是多少，因此，有

fi=max(fi,fj+ck+1,i+k)，j∈[1,i−1]f_i = max(f_i , f_j + c_{k + 1 , i} + k) ，j ∈[1 , i - 1] fi​=max(fi​,fj​+ck+1,i​+k)，j∈[1,i−1]

我们考虑 ck+1,ic_{k + 1 , i}ck+1,i​ 怎么求，显然 [j+1,i][j + 1 , i][j+1,i] 这几天使用最短路线是最优的

但是有的人可能会有疑惑，如果有那么一种情况，[l,r][l , r][l,r] 使用了一个路线，但是你将其划分成了两个区间，这岂不是变劣了？是这样的，因为 j∈[1,i−1]j ∈ [1 , i - 1]j∈[1,i−1]，所以我在枚举断点的时候一定会枚举到 [l,r][l , r][l,r] 使用一个路线的情况也就是说，这一次的转移终将被某一次正确的转移所覆盖掉

“不需要考虑每一次算的对不对，最优情况是否被算到”

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<queue>
#define int long long

using namespace std;

const int N = 1300;

int n , m , k , e , d; //
int l[N] , r[N] , p[N] , f[N];

inline int du(void){
    int x = 0 , f = 1;
    char ch = getchar();
    while(ch > '9' || ch < '0'){if(ch == '-')f = -1;ch = getchar();}
    while(ch >= '0' && ch <= '9'){x = x * 10 + ch - '0';ch = getchar();}
    return x * f;
}
inline void write(int x){
    if(x < 0){putchar('-');x = -x;}
    if(x > 9)write(x / 10);
    putchar(x % 10 ^ 48);
    return;
}

int head[N << 1] , ver[N << 1] , nx[N << 1] , w[N << 1] , idx;
inline void add(int x , int y , int z){
    ver[idx] = y , w[idx] = z;
    nx[idx] = head[x] , head[x] = idx ++ ;
    return;
}

int dis[N];
bool vis[N] , ok[N];
queue<int> que;

inline int spfa(int a , int b){
    memset(ok , true , sizeof(ok));
    for(int i = 1 ; i <= d ; ++ i){
        if((a <= l[i] && l[i] <= b) || (l[i] <= a && a <= r[i]))
            ok[p[i]] = false;
    }
    for(int i = 1 ; i <= m ; ++ i){
        dis[i] = (1ll << 50) , vis[i] = false;
    }
    dis[1] = 0 , vis[1] = true;
    que.push(1);
    while(que.size()){
        int u = que.front(); que.pop();
        vis[u] = false;
        for(int i = head[u] ; ~i ; i = nx[i]){
            int v = ver[i];
            if(ok[v] && dis[u] + w[i] < dis[v]){
                dis[v] = dis[u] + w[i];
                if(!vis[v]){
                    que.push(v);
                    vis[v] = true;
                }
            }
        }
    }
    return dis[m];
}

signed main(void){

    n = du() , m = du() , k = du() , e = du();

    memset(head , -1 , sizeof(head));

    for(int i = 1 ; i <= e ; ++ i){
        int x , y , z;
        x = du() , y = du() , z = du();
        add(x , y , z);
        add(y , x , z);
    }

    d = du();

    for(int i = 1 ; i <= d ; ++ i){
        p[i] = du() , l[i] = du() , r[i] = du();
    }

    for(int i = 1 ; i <= n ; ++ i){
        f[i] = (1ll << 50);
    }

    for(int i = 1 ; i <= n ; ++ i){
        f[i] = spfa(1 , i);
        for(int j = 1 ; j < i ; ++ j){
            f[i] = min(f[i] , f[j] + spfa(j + 1 , i) + k);
        }
    }

    write(f[n]);

    return 0;

}
