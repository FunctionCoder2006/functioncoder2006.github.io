Posted on 2021-10-16，5 min read

这道题首先我写的dfsdfsdfs思路很简单就是每次覆盖区间时尝试着尽可能多地覆盖区间。这个我不太会证明……但是想了想大概是这样的可能也不太严谨：对于我要解决的这个区间[L,R][L,R][L,R]我每用一个线段时我都尝试着让tatata尽可能多地覆盖区间，因为对于每一个元素，如果ta不被这个线段覆盖那ta就得被下一个线段覆盖。所以能让他被前面的区间覆盖就让前面的区间覆盖，后面的区间还省呢

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 1e5 + 10;

int n , a[N] , ans , cnt = 0;
int ansl[N] , ansr[N];

inline void dfs(int l , int r){

    if(l > r)return;

    int minn = (1 << 30);
    for(int i = l ; i <= r ; ++ i){
        minn = min(minn , a[i]);
    }

    for(int i = 1 ; i <= minn ; ++ i){
        ansl[++ cnt] = l;
        ansr[cnt] = r;
    }

    for(int i = l ; i <= r ; ++ i){
        a[i] -= minn;
    }

    ans += minn;

    int ql = l , qr = r;
    bool flag = false;

    for(int i = l ; i <= r ; ++ i){
        if(a[i] == 0){
            if(!flag){
                flag = true;  
                qr = i - 1;
            }
            else {
                ql = qr + 2;
                qr = i - 1;
            }
            dfs(ql , qr);
        }   
    }
    dfs(qr + 2 , r);
    return;
}

int main(void){

    //freopen("range.in" , "r" , stdin);
   // freopen("range.out" , "w" , stdout);

    scanf("%d" , &n);

    for(int i = 1 ; i <= n ; ++ i){
        scanf("%d" , &a[i]);
    }

    dfs(1 , n);

    printf("%d" , ans) , putchar('\n');

    for(int i = 1 ; i <= cnt ; ++ i){
        printf("%d %d\n" , ansl[i] , ansr[i]);
    }

    return 0;

}

如果我们在区间[L,R][L , R][L,R]上加1，就相当于在差分数组bbb上进行bL+=1,br+1−=1b_L += 1 , b_{r + 1} -= 1bL​+=1,br+1​−=1的操作（这里一定要注意是bLb_LbL​，还因为这个被学弟嘲笑了QAQ），如果区间是[i,n][i , n][i,n]的话只是在bbb数组上n+1n + 1n+1进行了−1- 1−1（因为b_{n + 1}是不做硬性要求的嘛，我们的目标状态是让是不做硬性要求的嘛，我们的目标状态是让是不做硬性要求的嘛，我们的目标状态是让b_1到b_n$变成目标值）

所以，我们肯定要先两个两个配对着对bbb数组进行变化。

但是考虑到∑i=1j\sum_{i = 1}^{j}∑i=1j​ = bj≥0b_j≥ 0bj​≥0 （其中j∈[1,n]j∈[1,n]j∈[1,n]）也就是说，bbb任意一个前缀都是恒非负的。所以我们可以知道最后一定是bbb数组正值偏多，最后很可能会留下几个正值，然后考虑我们不对bn+1b_{n + 1}bn+1​做硬性要求，所以bbb正值剩多少我就加几次

#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>

using namespace std;

const int N = 1e5 + 10;

int n , a[N] , b[N] , cnt1 , cnt2 , ansl[N] , ansr[N];

int main(void){

    scanf("%d" , &n);

    for(int i = 1 ; i <= n ; ++ i){
        scanf("%d" , &a[i]);
        b[i] = a[i] - a[i - 1];
    }

    for(int i = 1 ; i <= n ; ++ i){
        if(b[i] > 0){
            for(int k = 1 ; k <= b[i] ; ++ k)
                ansl[++ cnt1] = i;
        }
        if(b[i] < 0){
            for(int k = 1 ; k <= abs(b[i]) ; ++ k)
                ansr[++ cnt2] = i - 1;
        }
    }

    int tmp = cnt2 + 1;
    
    for(int i = tmp ; i <= cnt1 ; ++ i){
        ansr[++ cnt2] = n;
    }

    printf("%d\n" , cnt1);

    for(int i = 1 ; i <= cnt1 ; ++ i){
        printf("%d %d\n" , ansl[i] , ansr[i]);
    }

    return 0;

}
