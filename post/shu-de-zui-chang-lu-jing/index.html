树的最长路径

Posted on 2021-11-15，5 min read

考虑到 最长路径一定会中转某一个节点 uuu 那么中转点为 uuu 的最长路径是 d1+d2d1 + d2d1+d2 也就是子节点到 uuu 的最长路径和次长路径之和

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 1e5 + 10;

int n , ans , f[N] , d[N];

int head[N << 1] , nx[N << 1] , ver[N << 1] , w[N << 1] , idx;
inline void add(int x , int y , int z){
    ver[idx] = y , w[idx] = z;
    nx[idx] = head[x] , head[x] = idx ++ ;
    return;
}

inline void dfs(int u , int fa){
    int d1 = 0 , d2 = 0;
    for(int i = head[u] ; ~i ; i = nx[i]){
        int v = ver[i];
        if(v == fa)continue;
        dfs(v , u);
        if(f[v] + w[i] >= d1){
            d2 = d1;
            d1 = f[v] + w[i];
        }
        else if(f[v] + w[i] > d2){
            d2 = f[v] + w[i];
        }
    }
    f[u] = d1 , ans = max(ans , d1 + d2);
    return;
}

int main(void){

    scanf("%d" , &n);

    int x , y , z;

    memset(head , -1 , sizeof(head));

    for(int i = 1 ; i < n ; ++ i){
        scanf("%d%d%d" , &x , &y , &z);
        add(x , y , z);
        add(y , x , z);
    }

    dfs(1 , -1);

    printf("%d" , ans);

    return 0;

}

还有一种写法，其实也是求最长和次长，只是不是很明显，但是一定覆盖了最长 + 次长这种情况

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 1e5 + 10;

int n , ans , f[N] , d[N];

int head[N << 1] , nx[N << 1] , ver[N << 1] , w[N << 1] , idx;
inline void add(int x , int y , int z){
    ver[idx] = y , w[idx] = z;
    nx[idx] = head[x] , head[x] = idx ++ ;
    return;
}

inline void dfs(int u , int fa){
    for(int i = head[u] ; ~i ; i = nx[i]){
        int v = ver[i];
        if(v == fa)continue;
        dfs(v , u);
        ans = max(ans , d[u] + d[v] + w[i]);
        d[u] = max(d[u] , d[v] + w[i]);
    }
    return;
}

int main(void){

    scanf("%d" , &n);

    int x , y , z;

    memset(head , -1 , sizeof(head));

    for(int i = 1 ; i < n ; ++ i){
        scanf("%d%d%d" , &x , &y , &z);
        add(x , y , z);
        add(y , x , z);
    }

    dfs(1 , -1);

    printf("%d" , ans);

    return 0;

}

还有一种做法就是两次 bfsbfsbfs ，先随便选一个点 bfsbfsbfs，找到离 tatata 最远的点 uuu，然后再从 uuu 开始 bfsbfsbfs 找到离 uuu 最远的点 vvv ，u−>vu->vu−>v 便是最长路径

大体上就是证明 uuu 一定在直径的端点，然后从 uuu 开始跑 bfsbfsbfs 求出来的最大 dvd_vdv​ 就一定是直径

#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<queue>
#include<vector>

using namespace std;

const int N = 1e5 + 10;

int n , pre[N] , d[N];
bool st[N];

int head[N << 1] , ver[N << 1] , nx[N << 1] , w[N << 1] , idx;
inline void add(int x , int y , int z){
    ver[idx] = y , w[idx] = z;
    nx[idx] = head[x] , head[x] = idx ++ ;
    return;
}

queue<int> que;
vector<int> path;

inline int get_max(void){
    int t = -1 , maxn = 0;
    for(int i = 1 ; i <= n ; ++ i){
        if(d[i] > maxn){
            maxn = d[i];
            t = i;
        }
    }
    return t;
}

inline void bfs(int start){

    d[start] = 0;
    memset(pre , -1 , sizeof(pre));
    memset(st , false , sizeof(st));
    que.push(start);
    st[start] = true;

    while(que.size()){
        int u = que.front(); que.pop();
        for(int i = head[u] ; ~i ; i = nx[i]){
            int v = ver[i];
            if(st[v])continue;
            d[v] = d[u] + w[i];
            pre[v] = u;
            st[v] = true;
            que.push(v);
        }
    }

    return;

}

int main(void){

    scanf("%d" , &n);
    
    int x , y , z;
    
    memset(head , -1 , sizeof(head));

    for(int i = 1 ; i < n ; ++ i){
        scanf("%d%d%d" , &x , &y , &z);
        add(x , y , z);
        add(y , x , z);
    }

    bfs(1);

    int u = get_max();

    bfs(u);

    int v = get_max();

    printf("%d" , d[v]);

    return 0;
}
