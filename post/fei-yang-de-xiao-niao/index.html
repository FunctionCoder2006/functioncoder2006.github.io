这道题虽然很坑，但是还是能收获（得到）不少，可以说是一道好题

这道题主要是在考完全背包的方程推导，这道题特殊之处就是不能直接套用传统的完全背包，需要自己去推

先回忆完全背包的推导过程：

fi,j=max(fi−1,j,fi−1,j−wi+vi,fi−1,j−2wi+2vi,fi−1,j−3wi+3vi……)f_{i , j} = max(f_{i - 1 , j} , f_{i - 1 , j - w_i} + v_i , f_{i - 1 , j - 2w_i} + 2v_i , f_{i - 1 , j - 3w_i} + 3v_i……) fi,j​=max(fi−1,j​,fi−1,j−wi​​+vi​,fi−1,j−2wi​​+2vi​,fi−1,j−3wi​​+3vi​……)

fi,j−wi=max(fi−1,j−wi,fi−1,j−2wi+vi,fi−1,j−3wi+2vi……)f_{i , j - w_i} = max(f_{i - 1 , j - w_i} , f_{i - 1 , j - 2w_i} + v_i , f_{i - 1 , j - 3w_i} + 2v_i……) fi,j−wi​​=max(fi−1,j−wi​​,fi−1,j−2wi​​+vi​,fi−1,j−3wi​​+2vi​……)

然后发现 fi,j−wif_{i , j - w_i}fi,j−wi​​ 加上 viv_ivi​ 就变成了

max(fi−1,j−wi+vi,fi−1,j−2wi+2vi,fi−1,j−3wi+3vi……)max(f_{i - 1 , j - w_i} + v_i , f_{i - 1 , j - 2w_i} + 2v_i , f_{i - 1 , j - 3_wi} + 3v_i……) max(fi−1,j−wi​​+vi​,fi−1,j−2wi​​+2vi​,fi−1,j−3w​i​+3vi​……)

因此可以将原式子代换成 fi,j=max(fi−1,j,fi,j−wi+vi)f_{i , j} = max(f_{i - 1 , j} , f_{i , j - w_i} + v_i)fi,j​=max(fi−1,j​,fi,j−wi​​+vi​)

这道题也是一样的道理，设 fi,jf_{i , j}fi,j​ 表示所有位置到了 (i,j)(i , j)(i,j) 的方案中最小点击次数，所以，有

fi,j=min(fi−1,j+downi,fi−1,j−upi+1,fi−1,j−2upi+2,fi−1,j−3upi+3……)f_{i , j} = min(f_{i - 1 , j + down_i} , f_{i - 1 , j - up_i} + 1 , f_{i - 1 , j - 2up_i} + 2 , f_{i - 1 , j - 3up_i} + 3……) fi,j​=min(fi−1,j+downi​​,fi−1,j−upi​​+1,fi−1,j−2upi​​+2,fi−1,j−3upi​​+3……)

仍然考虑代换，但是我们对 fi−1,j−downif_{i - 1 , j - down_i}fi−1,j−downi​​ 束手无策，我们可以暂时不考虑 tatata，等算完了 fff 数组再取 minminmin 是没有影响的，因此，有

fi,j=min(fi−1,j−upi+1,fi,j−upi+1)f_{i , j} = min(f_{i - 1 , j - up_i} + 1 , f_{i , j - up_i} + 1) fi,j​=min(fi−1,j−upi​​+1,fi,j−upi​​+1)

还有一种情况，就是有可能点一下小鸟并不能飞 upiup_iupi​ 的高度，有可能被游戏界面阻挡了，需要特殊考虑。

依然考虑用代换的思想推导 fi,mf_{i , m}fi,m​ 的转移方程，有

fi,m=min(fi−1,k+1,fi,k+1)，k∈[m−upi,m]f_{i , m} = min(f_{i - 1 , k} + 1 , f_{i , k} + 1) ，k∈[m - up_i , m] fi,m​=min(fi−1,k​+1,fi,k​+1)，k∈[m−upi​,m]

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 1e4 + 10;
const int M = 1e3 + 10;

int n , m , k , up[N] , down[N];
int p[N] , l[N] , h[N] , low[N] , high[N];
int cnt[N] , f[N][M] , ans1 = (1 << 30) , ans2;
bool ok;

int main(void){

    scanf("%d%d%d" , &n , &m , &k);

    for(int i = 0 ; i < n ; ++ i){
        scanf("%d%d" , &up[i] , &down[i]);
    }//注意

    for(int i = 1 ; i <= n ; ++ i){
        low[i] = 0 , high[i] = m + 1;
    }

    for(int i = 0 ; i < k ; ++ i){
        scanf("%d%d%d" , &p[i] , &l[i] , &h[i]);
        low[p[i]] = l[i] , high[p[i]] = h[i];
        cnt[p[i]] ++ ;
    }

    for(int i = 1 ; i <= n ; ++ i){
        cnt[i] += cnt[i - 1]; 
    }

    memset(f , 0x3f , sizeof(f));

    for(int i = 1 ; i <= m ; ++ i){
        f[0][i] = 0;
    }

    for(int i = 1 ; i <= n ; ++ i){
        for(int j = up[i - 1] ; j <= m ; ++ j){
            f[i][j] = min(f[i - 1][j - up[i - 1]] + 1 , f[i][j - up[i - 1]] + 1);
            if(j == m){
                for(int k = m - up[i - 1] ; k <= m ; ++ k){
                    f[i][m] = min(f[i][m] , min(f[i - 1][k] + 1 , f[i][k] + 1));
                }
            }
        }
        for(int j = 0 ; j + down[i - 1] <= m ; ++ j){
            f[i][j] = min(f[i][j] , f[i - 1][j + down[i - 1]]);
        }
        for(int j = high[i] ; j <= m ; ++ j){
            f[i][j] = f[0][0]; 
        }
        for(int j = 0 ; j <= low[i] ; ++ j){
            f[i][j] = f[0][0];
        }
    }

    for(int i = 1 ; i <= m ; ++ i){
        if(f[n][i] < f[0][0]){
            ok = true;
            ans1 = min(ans1 , f[n][i]);
        }
    }

    if(ok){
        puts("1");
        printf("%d" , ans1);
        return 0;
    }

    puts("0");

    for(int i = n ; i >= 0 ; -- i){
        for(int j = 1 ; j <= m ; ++ j){
            if(f[i][j] < f[0][0]){
                printf("%d" , cnt[i]);
                return 0;
            }
        }
    }

    return 0;

}
