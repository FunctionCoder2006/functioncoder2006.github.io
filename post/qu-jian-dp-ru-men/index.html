石子合并

区间 dpdpdp 经典题了

主要讨论一下区间 dpdpdp 和线性区间划分的区别

做了这么多区间划分题了，应该不难发现有连续的区间性质

首先我们考虑区间划分，设 fif_ifi​ 表示将前 iii 个石子合并能够得到地最大值，我们考虑将区间进行划分，我们设断点为 jjj，那么，有

fi=max(fi,fj+cost(j+1,i))，j∈[1,i−1]f_i = max(f_i , f_j + cost(j + 1 , i))，j ∈[1 , i - 1] fi​=max(fi​,fj​+cost(j+1,i))，j∈[1,i−1]

但是我们很快发现 cost(j+1,i)cost(j + 1 , i)cost(j+1,i) 没有办法直接计算，必须对 [j+1,i][j + 1 , i][j+1,i] 这个区间继续进行划分

因此，区间 dpdpdp 就诞生了

设 fl,rf_{l , r}fl,r​ 表示将区间 [l,r][l , r][l,r] 的石子进行合并得到的最大值（或者说将区间 [l,r][l , r][l,r] 进行区间划分得到的最大值），枚举端点 kkk，因此，有

fl,r=max(fl,r,fl,k+fk+1,r+∑i=lnai)f_{l , r} = max(f_{l , r} , f_{l , k} + f_{k + 1 , r} + \sum_{i = l}^{n}a_i) fl,r​=max(fl,r​,fl,k​+fk+1,r​+i=l∑n​ai​)

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 110;

int n , f[N][N] , ans = -(1 << 30) , a[N] , g[N][N];
char s[N] , ins[N][N];

int main(void){

    cin >> n;

    for(int i = 1 ; i <= n ; ++ i){
        cin >> s[i] >> a[i];
        if(i != 1){
            ins[i - 1][i] = s[i];
        }
        ins[i + n - 1][i + n] = s[i];
        a[i + n] = a[i];
    }

    memset(f , ~0x3f , sizeof(f));
    memset(g , 0x3f , sizeof(g));

    for(int i = 1 ; i <= n * 2 ; ++ i){
        f[i][i] = a[i];
        g[i][i] = a[i];
    }

    for(int Len = 2 ; Len <= n * 2 ; ++ Len){
        for(int l = 1 ; l + Len - 1 <= n * 2 ; ++ l){
            int r = l + Len - 1;
            for(int k = l ; k < r ; ++ k){
                int tmp;
                if(ins[k][k + 1] == 'x'){
                    tmp = max(f[l][k] * f[k + 1][r] , g[l][k] * g[k + 1][r]);
                }
                else tmp = f[l][k] + f[k + 1][r];
                f[l][r] = max(f[l][r] , tmp);
                if(ins[k][k + 1] == 't'){
                    tmp = g[l][k] + g[k + 1][r];
                }
                else tmp = min(f[l][k] * g[k + 1][r] , g[l][k] * f[k + 1][r]);
                g[l][r] = min(g[l][r] , tmp);
            }
        }
    }

    for(int i = 1 ; i <= n ; ++ i){
        ans = max(ans , f[i][i + n - 1]);
    }

    cout << ans << endl;

    for(int i = 1 ; i <= n ; ++ i){
        if(f[i][i + n - 1] == ans){
            cout << i << " " ;
        }
    }

    return 0;

}
