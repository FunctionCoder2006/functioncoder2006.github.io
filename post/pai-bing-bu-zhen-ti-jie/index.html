2021-11-11 2 min read # 背包 # 动态规划

分组背包

额外需要注意的是预处理每一个组的“物品”的代价， 有一些状态是冗余的。

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 110;
const int M = 2e4 + 10;

int s , n , m , a[N][N] , c[N][N];
int f[M];

int main(void){
    
    scanf("%d%d%d" , &s , &n , &m);

    for(int i = 1 ; i <= s ; ++ i){
        for(int j = 1 ; j <= n ; ++ j){
            scanf("%d" , &a[i][j]);
            c[j][i] = a[i][j];
        }
    }

    for(int i = 1 ; i <= n ; ++ i){
        sort(c[i] + 1 , c[i] + 1 + s);
        for(int j = 1 ; j <= s ; ++ j){
            c[i][j] <<= 1;
            c[i][j] ++;
        }
    }

    for(int i = 1 ; i <= n ; ++ i){
        for(int j = m ; j >= 0 ; -- j){
            for(int k = 0 ; k <= s ; ++ k){
                if(c[i][k] > j)break;
                f[j] = max(f[j] , f[j - c[i][k]] + k * i);
            }
        }
    }

    printf("%d" , f[m]);

    return 0;

}
