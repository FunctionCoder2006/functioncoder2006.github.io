乌龟棋

Posted on 2021-12-22，2 min read

这道题其实挺简单的，和杨老师的照相排列那道题差不多

需要强化的是，在读题的时候应该留意有哪些数据是特殊的。对于这道题，我们发现只有四种牌并且每一种牌不超过 404040 个并且每次只会用一个牌

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

const int N = 360;
const int M = 43;

int n , m , a[N] , b[N] , f[M][M][M][M] , cnt[M];

int main(void){

    cin >> n >> m;

    for(int i = 1 ; i <= n ; ++ i){
        cin >> a[i];
    }

    for(int i = 1 ; i <= m ; ++ i){
        cin >> b[i];
        cnt[b[i]] ++ ;
    }

    f[0][0][0][0] = a[1];

    for(int i = 0 ; i <= cnt[1] ; ++ i)
        for(int j = 0 ; j <= cnt[2] ; ++ j)
            for(int k = 0 ; k <= cnt[3] ; ++ k)
                for(int p = 0 ; p <= cnt[4] ; ++ p){
                    int state = i + j * 2 + k * 3 + p * 4;
                    f[i + 1][j][k][p] = max(f[i + 1][j][k][p] , f[i][j][k][p] + a[state + 1 + 1]);
                    f[i][j + 1][k][p] = max(f[i][j + 1][k][p] , f[i][j][k][p] + a[state + 2 + 1]);
                    f[i][j][k + 1][p] = max(f[i][j][k + 1][p] , f[i][j][k][p] + a[state + 3 + 1]);
                    f[i][j][k][p + 1] = max(f[i][j][k][p + 1] , f[i][j][k][p] + a[state + 4 + 1]);
                }

    cout << f[cnt[1]][cnt[2]][cnt[3]][cnt[4]] << endl;

    return 0;

}
