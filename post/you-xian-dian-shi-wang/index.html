2021-11-15 4 min read # 树形dp # 背包 # 动态规划

这道题乍一看用户都在叶子节点，感觉无法转移

总感觉如果将花费存入状态不太妙，况且题目中并没有明确给出花费的具体范围

因为分配给用户的数量一定时，花费越少越有可能不赔本，所以考虑将分配用户的数量存入状态，
设 fu,i,jf_{u , i , j}fu,i,j​ 表示以 uuu 为子树，只考虑前 iii 个子树，已经分配了 jjj 个用户的最小花费。考虑给第 iii 个子树分配 ttt 个用户，有

fu,i,j=min(fu,i,j,fu,i−1,j−t+fv,sub[v],t+w[i])，t∈[0,min(j,siz[v])]f_{u , i , j} = min(f_{u , i , j} , f_{u , i - 1 , j - t} + f_{v , sub[v] , t} + w[i])，t ∈ [0 , min(j , siz[v])] fu,i,j​=min(fu,i,j​,fu,i−1,j−t​+fv,sub[v],t​+w[i])，t∈[0,min(j,siz[v])]

其中，sub[v]sub[v]sub[v] 和 siz[v]siz[v]siz[v] 分别表示以 vvv 有多少子树和以 vvv 为子树，有多少用户即有多少叶节点

考虑滚动数组优化，因此有

fu,j=min(fu,j,fu,j−t+fv,t+w[i])，t∈[0,min(j,siz[v])]f_{u , j} = min(f_{u , j} , f_{u , j - t} + f_{v , t} + w[i])，t ∈ [0 , min(j , siz[v])] fu,j​=min(fu,j​,fu,j−t​+fv,t​+w[i])，t∈[0,min(j,siz[v])]

这里和选课一题不一样的是这题可以将所有用户分配到一个子树里

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
#include<cstring>

using namespace std;

const int N = 3010;

int n , m , f[N][N] , siz[N];

int head[N << 1] , nx[N << 1] , ver[N << 1] , idx , w[N << 1];
inline void add(int x , int y , int z){
    ver[idx] = y , w[idx] = z;
    nx[idx] = head[x] , head[x] = idx ++ ;
    return;
}

inline void dfs(int u , int fa){
    f[u][0] = 0;
    for(int i = head[u] ; ~i ; i = nx[i]){
        int v = ver[i];
        if(v == fa)continue;
        dfs(v , u);
        siz[u] += siz[v];
        for(int j = min(m , siz[u]) ; j >= 0 ; -- j){
            for(int t = min(siz[v] , j) ; t >= 0 ; -- t){
                int c = w[i];
                if(t == 0)c = 0;
                f[u][j] = min(f[u][j] , f[u][j - t] + f[v][t] + c);
            }
        }
    }
    return;
}

int main(void){

    scanf("%d%d" , &n , &m);

    memset(head , -1 , sizeof(head));

    memset(f , 0x3f , sizeof(f));

    for(int i = 1 ; i <= n - m ; ++ i){
        int k , x , c;
        scanf("%d" , &k);
        for(int j = 1 ; j <= k ; ++ j){
            scanf("%d%d" , &x , &c);
            add(i , x , c);
            add(x , i , c);
        }
    }

    for(int i = n - m + 1 ; i <= n ; ++ i){
        int c;
        scanf("%d" , &c);
        siz[i] = 1;
        f[i][1] = -c;
    }

    dfs(1 , -1);

    for(int j = m ; j >= 0 ; -- j){
        if(f[1][j] <= 0){
            printf("%d" , j);
            return 0;
        }
    }


    puts("0");

    return 0;
}
